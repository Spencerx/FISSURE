# Define a command alias for all FISSURE-related commands
Cmnd_Alias FISSURE_CMDS = /usr/bin/killall gpsd, \
                           /usr/bin/fuser, \
                           /usr/sbin/gpsd, \
                           /sbin/ifconfig, \
                           /sbin/iwconfig

# Allow the user to run all commands in the FISSURE_CMDS alias without a password
user ALL=(ALL) NOPASSWD: FISSURE_CMDS
