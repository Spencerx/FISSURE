
# Build a Writable Sandbox
sudo apptainer build --sandbox fissure_apptainer fissure_apptainer.def

# Ensure cookie exists and copy to a neutral path
if [ -f "$HOME/.config/pulse/cookie" ]; then
    cp "$HOME/.config/pulse/cookie" /tmp/pulse-cookie
    chmod 644 /tmp/pulse-cookie
else
    echo "[!] PulseAudio cookie not found for $USER"
fi

# Enter Writable Apptainer with GUI Support (X11)
echo $XDG_SESSION_TYPE
sudo apt-get install -y apptainer xauth x11-apps x11-utils x11-common libx11-6 libxext6
sudo apt-get install -y fuse-overlayfs
xhost +local:$USER
sudo apptainer shell --writable --bind /tmp/.X11-unix:/tmp/.X11-unix --env DISPLAY=$DISPLAY fissure_apptainer
sudo apptainer shell \
  --writable \
  --bind /dev/bus/usb:/dev/bus/usb \
  $(for dev in /dev/ttyACM* /dev/ttyUSB*; do [ -e "$dev" ] && echo --bind "$dev:$dev"; done) \
  --bind /run/udev:/run/udev \
  --bind /tmp/.X11-unix:/tmp/.X11-unix \
  --bind /run/user/$(id -u $SUDO_USER)/pulse:/tmp/pulse \
  --env DISPLAY=$DISPLAY,PULSE_SERVER=unix:/tmp/pulse/native,PULSE_COOKIE=/tmp/pulse-cookie \
  --env XDG_RUNTIME_DIR=/tmp/runtime-root \
  ~/fissure_apptainer

# Enter Writable Apptainer with GUI Support (Wayland)
sudo apt-get install -y apptainer xwayland xauth x11-apps
sudo apt-get install -y fuse-overlayfs
xhost +si:localuser:root
sudo apptainer shell \
  --writable \
  --bind /tmp/.X11-unix:/tmp/.X11-unix \
  --bind /run/user/$(id -u $SUDO_USER)/pulse:/tmp/pulse \
  --env DISPLAY=$DISPLAY,PULSE_SERVER=unix:/tmp/pulse/native,PULSE_COOKIE=/tmp/pulse-cookie \
  --env QT_QPA_PLATFORM=xcb \
  --env GDK_BACKEND=x11 \
  fissure_apptainer

#sudo apt-get install -y x11-apps xwayland libx11-6  # Wayland (on host and in apptainer?)

# Exit Apptainer
exit

# Delete Apptainer
sudo rm -Rf fissure_apptainer
